# Offline docling extraction tool (permitted per safaris-knowledgebase ADR-0003 clause 4).
# Pinned so the script produces the same artifacts whenever/whoever runs it.
# docling pulls its own heavy deps (torch, pandas, transformers, …).
docling==2.108.0

# OCR for scanned/image-only PDFs. docling 2.108's default (RapidOCR/torch) is
# broken ("Unsupported configuration: torch.PP-OCRv6"). extract.py prefers Apple
# Vision (ocrmac) on macOS — best on clean scans — and falls back to EasyOCR
# elsewhere. Both download/use their models locally.
ocrmac==1.0.1 ; sys_platform == "darwin"
easyocr==1.7.2
